CSS — Applying the school theme
CSS + JS
/* CSS variables swap at login — never hardcode school color */
:root {
  --p:      #1A6BB5;   /* school primary */
  --p-tint: rgba(26,107,181,0.08);
  --p-mid:  rgba(26,107,181,0.18);
}

/* JS — call this after login */
function applyTheme(school) {
  const cfg = SCHOOL_CFG[school];
  document.documentElement.style.setProperty('--p',      cfg.primary);
  document.documentElement.style.setProperty('--p-tint', cfg.tint);
  document.documentElement.style.setProperty('--p-mid',  cfg.mid);
}

06 — Motion

All transitions use one shared easing curve. Durations are short and purposeful.

Easing
CSS
--ease: cubic-bezier(0.16, 1, 0.3, 1);  /* fast out, spring-like */
--t:    0.45s;  /* school theme color transition */

DurationUsage

/* JS — drag handler pattern */
fc.addEventListener('pointermove', e => {
  const dx = e.clientX - startX;
  fc.style.transform = `translateX(${dx}px) rotate(${dx * .05}deg)`;
  const pct = Math.min(Math.abs(dx) / 80, 1);
  saveLbl.style.opacity = dx > 0 ? pct : 0;
  skipLbl.style.opacity = dx < 0 ? pct : 0;
});

17 — Map Markers

Custom Leaflet div icons. Teardrop shape via border-radius: 50% 50% 50% 0 rotated −45°. Active marker is larger with stronger shadow.

default 28px
active 34px
JavaScript
function makeIcon(color, active) {
  const sz = active ? 34 : 28;
  return L.divIcon({
    className: '',
    html: `<div style="
      width:${sz}px; height:${sz}px;
      border-radius: 50% 50% 50% 0;
      background: ${color};
      border: ${active ? '3px' : '2px'} solid white;
      box-shadow: 0 ${active ? 6 : 3}px ${active ? 18 : 10}px
                  rgba(0,0,0,${active ? .28 : .18});
      transform: rotate(-45deg);
      transition: all .25s;
    "></div>`,
    iconSize:    [sz, sz],
    iconAnchor:  [sz/2, sz],
    popupAnchor: [0, -sz - 4],
  });
}
